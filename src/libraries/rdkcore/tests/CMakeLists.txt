MACRO(RDK_ADD_TEST testname)
	ADD_EXECUTABLE(${testname} ${testname}.cpp)
	IF(RDK_INSIDE_OpenRDK)
		SET(OUTDIR "${OpenRDK_OUTPUT_TREE}/bin/tests")
	ELSE(RDK_INSIDE_OpenRDK)
		SET(OUTDIR "${OpenRDK_OUTPUT_TREE}/binext/tests")
	ENDIF(RDK_INSIDE_OpenRDK)
	TARGET_LINK_LIBRARIES(${testname} rdkcore)
	SET_TARGET_PROPERTIES(${testname} PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${OUTDIR}")
	ADD_TEST(${testname} ${OUTDIR}/${testname})
ENDMACRO(RDK_ADD_TEST)

RDK_ADD_TEST(unittest_url)
